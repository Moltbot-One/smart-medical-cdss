## 1. 概述

### 1.1 系统简介

智慧医疗临床决策支持系统(CDSS)是基于人工智能的临床辅助决策平台，为医疗机构提供：

- **临床决策支持：** 基于循证医学的知识库推理
- **药物相互作用检测：** 实时处方审核与用药建议
- **诊疗路径推荐：** 个性化临床路径管理
- **医学知识检索：** 结构化医学知识图谱查询
- **质量指标监控：** 医疗质量实时监测与预警

### 1.2 部署范围

| 组件 | 说明 |
|------|------|
| CDSS Core | 核心业务服务（推理引擎、知识库服务） |
| CDSS Gateway | API网关与流量管理 |
| CDSS Web | 前端Web应用 |
| CDSS Admin | 管理后台 |
| Knowledge Base | 医学知识库服务 |
| AI/ML Services | 模型推理服务（NLP、CV） |
| Data Platform | 数据集成与分析平台 |

### 1.3 目标读者

- DevOps工程师
- 系统架构师
- 运维工程师
- 安全工程师
- 医疗信息化项目经理

#### 6.1.1 基础设施预检清单

```bash
#!/bin/bash
# preflight-check.sh - 部署前环境检查脚本

set -euo pipefail

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 日志函数
log_info() { echo -e "${GREEN}[INFO]${NC} $1"; }
log_warn() { echo -e "${YELLOW}[WARN]${NC} $1"; }
log_error() { echo -e "${RED}[ERROR]${NC} $1"; }

# 检查结果计数器
CHECKS_PASSED=0
CHECKS_FAILED=0

# 检查函数
check_pass() {
    log_info "✓ $1"
    ((CHECKS_PASSED++))
}

check_fail() {
    log_error "✗ $1"
    ((CHECKS_FAILED++))
}

echo "======================================"
echo "CDSS平台部署前环境检查"
echo "======================================"
echo ""

# 1. 系统要求检查
echo "【1/10】系统要求检查..."

# 1.1 操作系统
if [[ -f /etc/os-release ]]; then
    source /etc/os-release
    if [[ "$ID" == "ubuntu" && "$VERSION_ID" == "22.04" ]] || \
       [[ "$ID" == "centos" && "$VERSION_ID" == "7" ]] || \
       [[ "$ID" == "rocky" && "$VERSION_ID" == "8" ]]; then
        check_pass "操作系统: $NAME $VERSION_ID"
    else
        check_fail "操作系统版本不符合要求 (需要Ubuntu 22.04/CentOS 7/Rocky 8)"
    fi
else
    check_fail "无法检测操作系统版本"
fi

# 1.2 内核版本
KERNEL_MAJOR=$(uname -r | cut -d. -f1)
KERNEL_MINOR=$(uname -r | cut -d. -f2)
if [[ $KERNEL_MAJOR -ge 5 && $KERNEL_MINOR -ge 15 ]] || [[ $KERNEL_MAJOR -ge 6 ]]; then
    check_pass "内核版本: $(uname -r)"
else
    check_fail "内核版本过低 (需要 >= 5.15)"
fi

# 1.3 硬件资源
CPU_CORES=$(nproc)
MEMORY_GB=$(free -g | awk '/^Mem:/{print $2}')
DISK_GB=$(df -BG / | awk 'NR==2{gsub(/G/,"",$2); print $2}')

if [[ $CPU_CORES -ge 16 ]]; then
    check_pass "CPU核心数: ${CPU_CORES}"
else
    check_fail "CPU核心数不足 (需要 >= 16)"
fi

if [[ $MEMORY_GB -ge 64 ]]; then
    check_pass "内存容量: ${MEMORY_GB}GB"
else
    check_fail "内存容量不足 (需要 >= 64GB)"
fi

if [[ $DISK_GB -ge 500 ]]; then
    check_pass "系统磁盘: ${DISK_GB}GB"
else
    check_fail "系统磁盘空间不足 (需要 >= 500GB)"
fi

# 2. 网络检查
echo ""
echo "【2/10】网络连接检查..."

# 2.1 主机名解析
if [[ -n $(hostname -f) ]]; then
    check_pass "主机名配置: $(hostname -f)"
else
    check_fail "主机名未正确配置"
fi

# 2.2 DNS解析
if nslookup google.com > /dev/null 2>&1; then
    check_pass "DNS解析正常"
else
    check_fail "DNS解析失败"
fi

# 2.3 外网连接
if ping -c 1 -W 5 8.8.8.8 > /dev/null 2>&1; then
    check_pass "外网连接正常"
else
    check_warn "外网连接受限 (可能需要代理)"
fi

# 2.4 NTP同步
if command -v timedatectl &> /dev/null; then
    NTP_STATUS=$(timedatectl show --property=NTP --value 2>/dev/null || echo "no")
    if [[ "$NTP_STATUS" == "yes" ]]; then
        check_pass "NTP时间同步已启用"
    else
        check_fail "NTP时间同步未启用"
    fi
else
    check_warn "无法检测NTP状态"
fi

# 3. 存储检查
echo ""
echo "【3/10】存储配置检查..."

# 3.1 磁盘空间
DISK_USAGE=$(df -h / | awk 'NR==2{gsub(/%/,"",$5); print $5}')
if [[ $DISK_USAGE -lt 80 ]]; then
    check_pass "根分区使用率: ${DISK_USAGE}%"
else
    check_fail "根分区使用率过高: ${DISK_USAGE}%"
fi

# 3.2 磁盘I/O调度器
if [[ -f /sys/block/sda/queue/scheduler ]]; then
    SCHEDULER=$(cat /sys/block/sda/queue/scheduler | grep -oE '\[.*\]' | tr -d '[]')
    if [[ "$SCHEDULER" == "none" ]] || [[ "$SCHEDULER" == "mq-deadline" ]]; then
        check_pass "磁盘I/O调度器: $SCHEDULER"
    else
        check_warn "建议将I/O调度器设置为none或mq-deadline (NVMe)"
    fi
fi

# 4. 安全基线检查
echo ""
echo "【4/10】安全基线检查..."

# 4.1 SELinux/AppArmor
if command -v getenforce &> /dev/null; then
    SELINUX_STATUS=$(getenforce)
    if [[ "$SELINUX_STATUS" == "Enforcing" ]] || [[ "$SELINUX_STATUS" == "Permissive" ]]; then
        check_pass "SELinux状态: $SELINUX_STATUS"
    else
        check_warn "SELinux已禁用 (建议启用)"
    fi
elif command -v aa-status &> /dev/null; then
    check_pass "AppArmor已安装"
else
    check_warn "未检测到MAC系统 (SELinux/AppArmor)"
fi

# 4.2 防火墙
if command -v ufw &> /dev/null; then
    UFW_STATUS=$(ufw status | head -1)
    if [[ "$UFW_STATUS" == *"active"* ]]; then
        check_pass "UFW防火墙: 已启用"
    else
        check_warn "UFW防火墙: 未启用"
    fi
elif command -v firewall-cmd &> /dev/null; then
    if firewall-cmd --state 2>/dev/null | grep -q "running"; then
        check_pass "Firewalld: 已启用"
    else
        check_warn "Firewalld: 未启用"
    fi
fi

# 5. 容器运行时检查
echo ""
echo "【5/10】容器运行时检查..."

# 5.1 Containerd
if command -v containerd &> /dev/null; then
    CONTAINERD_VERSION=$(containerd --version | head -1 | awk '{print $3}')
    check_pass "Containerd版本: $CONTAINERD_VERSION"
else
    check_fail "未安装Containerd"
fi

# 5.2 runc
if command -v runc &> /dev/null; then
    RUNC_VERSION=$(runc --version | head -1 | awk '{print $3}')
    check_pass "runc版本: $RUNC_VERSION"
else
    check_fail "未安装runc"
fi

# 6. Kubernetes工具检查
echo ""
echo "【6/10】Kubernetes工具检查..."

# 6.1 kubectl
```

```bash
  -t cdss.cdss_decision_logs \
  --format=custom \

    aws s3 rm s3://cdss-backups/postgresql/${dir_name} --recursive
  fi

    "cdss.cdss_decision_logs"
```
